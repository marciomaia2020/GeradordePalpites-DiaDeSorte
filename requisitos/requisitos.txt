        const modalidades = [
            { nome: "Dia de Sorte", intervalo: 31 },
            { nome: "Lotofácil", intervalo: 25 },
            { nome: "Super 7", intervalo: 70 },
            { nome: "Lotomania", intervalo: 100 },
            { nome: "Dupla Sena", intervalo: 50 },
            { nome: "Quina", intervalo: 80 },
            { nome: "Timemania", intervalo: 80 },
            { nome: "Mega-Sena", intervalo: 60 }
        ];

        function calcularCombinacoes(digitos, intervalo) {
            let min = Math.pow(10, digitos - 1); // Menor número com X dígitos
            let max = Math.pow(10, digitos) - 1; // Maior número com X dígitos
            let validos = 0;

            for (let i = min; i <= max; i++) {
                if (i <= intervalo) {
                    validos++;
                }
            }
            return validos;
        }

        const tabelaCombinacoes = document.getElementById("tabela-combinacoes");

        modalidades.forEach(modalidade => {
            const row = document.createElement("tr");
            row.innerHTML = `
                <td>${modalidade.nome}</td>
                <td>01-${modalidade.intervalo}</td>
                <td>${calcularCombinacoes(1, modalidade.intervalo)}</td>
                <td>${calcularCombinacoes(2, modalidade.intervalo)}</td>
                <td>${calcularCombinacoes(3, modalidade.intervalo)}</td>
                <td>${calcularCombinacoes(4, modalidade.intervalo)}</td>
                <td>${calcularCombinacoes(5, modalidade.intervalo)}</td>
                <td>${calcularCombinacoes(6, modalidade.intervalo)}</td>

                <td>${calcularCombinacoes(8, modalidade.intervalo)}</td>
                <td>${calcularCombinacoes(9, modalidade.intervalo)}</td>
            `;
            tabelaCombinacoes.appendChild(row);
        });
    </script>
